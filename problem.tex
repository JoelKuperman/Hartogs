\section{Preliminaries}

\begin{definition}
	We shall use $\estf$ (Elementary Set Theory) to mean the axioms of Extensionality, Comprehension, Pairing, Union and Powerset.
\end{definition}

Throughout this section we will be working within $\estf$ unless otherwise specified.
Without Replacement nor Choice it is consistent for several common definitions
of infinite set to not be equivalent to each other, (see Dedekind finite sets,
amorphous sets, etc.) which motivates the addition of a weaker version of the
Axiom of Infinity.

\begin{framed}
  One other axiom that might be considered affected/involved with uses of
  Replacement is the Axiom of Infinity. As our intent is to work as naïvely
  (i.e. as the working mathematician's intuitive idea of set) as possible, we
  prefer to consider a version of this axiom that does not involve ordinals in
  any implicit sense. Thus, we are not stating Infinity as the existence of
  $\omega$. For instance, it is a well known fact that Zermelo's
  \textit{Unendlichen} axiom does not imply Von Neumann's one
  without Replacement (e.g., \cite[Exercise II.4.21]{Kunen}), and the lack of $\AC$ brings even more elements to the
  picture (infinite Dedekind finite sets, amorphous sets, etc.).
\end{framed}

\fbox{To fix notation\dots} Before stating this axiom, let us recall the following definitions for ease of reference.

\begin{definition}
	\begin{enumerate}
		\item A set $n$ is a \emph{natural number} iff it is the empty set or a successor ordinal and for each $m\in n$, $m$ is the empty set or a successor ordinal.
		\item A set $A$ is \emph{finite} iff $A\leq_c n$ for some natural number $n$.
		\item A set is \emph{infinite} iff it is not finite.
	\end{enumerate}
\end{definition}
In the following, we write “$n\in\omega$” for the formula defining natural numbers:
\[
\forall x, \, x \in n \lor  x = n \implies x = \emptyset \lor (\exists y \in
n, \, x = y \cup \{y\}).
\]

\begin{WeakInf}
	There is an infinite set.
	%There is a well ordering $\langle A , < \rangle$ such that for all $x\in A$, $x$ has an immediate successor.
\end{WeakInf}

We shall use $\esti$ to denote $\estf$ with the Weak Axiom of Infinity.
Note that it is consistent with $\ZC$ minus the standard Axiom of Infinity that
an infinite well order  exists and for the Axiom of Infinity to fail
(\cite[Exercise II.4.21]{Kunen} gives instructions on how to build such a
model), in particular $\esti$ is consistent with the negation of Infinity relative to ZFC.

\begin{definition}
  For sets $X,Y$ and  linear orders $V$ and $U$,
  \begin{enumerate}
  \item $X\leq_c Y$ iff there is an injective function $f:X\to Y$, and $X =_c Y$ iff there is a bijective function $f:X\to Y$.
  \item
    we write $V\leq_o U$ iff  $V$ is
    isomorphic to an initial segment of $U$;
  \item we use $V<_o U$ when  $V\leq_o U$ and  $V$ is not order isomorphic
    to $U$.
  \end{enumerate}
\end{definition}

The following proposition is provable in $\estf$ and is necessary to prove some elementary facts about finite sets using induction over $\omega$, which may be a proper class even in $\esti$.

\begin{prop}[{\cite[Corollary 7.33]{Moscho}}]\label{class_wo}
  Every non-empty class $\mathcal{C}$ of well ordered sets has a $\leq_o$-least member.
\end{prop}

We shall start by laying out a few well known definitions and propositions about the Hartogs function. Of particular note is the bound $\aleph(X) <_c \mathcal{P}^{3}(X)$, which is the key idea for our proof of Theorem \ref{informal_hartogs_set}.

\begin{definition}\label{aleph}
	Let $X$ be any set,
	\[
	\WO(X):=\{\langle A,R\rangle \mid A\subset X \ \text{is well-ordered by}\ R\}, 
	\]
	and $\aleph(X):= \WO(X)/{\cong}$ where $\cong$ denotes order isomorphism.
        %% $\langle A, R\rangle =_o \langle B,  R'\rangle$ iff there exists an  \emph{initial similarity}
        %% \cite[Def.~7.27]{Moscho} between those two well orders.
	Given $\aleph(X)$, and $A,B \in\aleph(X)$ we shall say $A \leq_{\aleph(X)} B$ iff for any (all) $U\in A$ and $V\in B$ it holds that $U\leq_o V$.
\end{definition}

From Moschovakis \cite[Thm.~7.34]{Moscho} we conclude
\begin{lemma}\label{aleph2}
	For all $X$, $\aleph(X)$ has the following properties:
	\begin{enumerate}
		\item $\aleph(X)\nleq_c X$.
		\item $(\aleph(X),\leq_{\aleph(X)})$ is a well order and is not
                  equinumerous with any of its proper initial segments.
		\item If $W$ is a well ordered set and $W \nleq_c X$, then $\aleph(X) \leq_o W$.\qed
	\end{enumerate}
\end{lemma}

\begin{corollary}\label{cor:ltc-iff-lto}
  $\aleph(X) <_c \aleph(Y)$ if and only if $\aleph(X) <_o \aleph(Y)$.
\end{corollary}
\begin{proof}
  It follows by Lemma~\ref{aleph2} and the fact that the relation $\leq_o$ is
  total (Proposition~\ref{class_wo}, or more precisely \cite[Thm.~7.31]{Moscho}).
\end{proof}

\begin{theorem}\label{thm:partes-cubo}
	For any set $X$, $\aleph(X) <_c \mathcal{P}^3(X)$.
	%$\aleph(X)<_c \mathcal{P}^3(X)$.
\end{theorem}
\begin{proof}	
  Let $f:\WO(X)\rightarrow \mathcal{P}^2(X)$ defined by $f(\langle
  A,\mathrel{R}\rangle) = \{ \{ b\mid b \mathrel{R} a \} \mid a\in A \}$. $f$
  is clearly injective. Hence, the injective function
  $\Tilde{f}: \mathcal{P} (\WO(X)) \rightarrow \mathcal{P}^3(X)$ given
  by $\Tilde{f}(S) = \{ f(s) \mid s\in S \}$ witnesses
  $\mathcal{P} (\WO(X)) \leq_c \mathcal{P}^3(X)$. Since
  $\aleph (X) \subset    \mathcal{P} (\WO(X))$ we conclude
  $\aleph(X) \leq_c \mathcal{P}^3(X)$.
            
	Let us prove that $\aleph(X) \neq_c \mathcal{P}^3(X)$. 
	Proceding by contradiction, we assume they are equinumerous, then $\mathcal{P}^{3}(X)$ is well orderable. As an immediate consequence we have that $\mathcal{P}^{2}(X)$ is well orderable. But then $\mathcal{P}^{2}(X)$ is a well orderable set and it satisfies 	
	$\mathcal{P}^{2}(X)\nleq_c X$.
	So Lemma \ref{aleph2} implies that 	
	$\aleph(X)\leq_c\mathcal{P}^{2}(X)$.
	But 	
	$\mathcal{P}^{3}(X) =_c \aleph(X)$	
	by assumption, so we have shown $\mathcal{P}^{3}(X) \leq_c \mathcal{P}^{2}(X)$, which gives us our contradiction.
\end{proof}

see also the discussion at \cite{stack_powerset}.

Using the wellfoundedness of  $\leq_o$, we verify that Induction works on
$\estf$. This will be later needed for ...
\begin{theorem}
	Let $P(n,\vec{x})$ be a first order formula in the language of set theory whose free variables are either $n$ or contained in $\vec{x}$. Assume that the following statements hold:
	\begin{enumerate}[(i)]
		\item $P(\emptyset,\vec{x})$.
		\item\label{inductive} Let $n$ be a natural number, then $P(n,\vec{x}) \implies P(n\cup \{n\},\vec{x})$.
	\end{enumerate}
	Then $P(n,\vec{x})$ is true for every natural number.
\end{theorem}
\begin{proof}
	By way of contradiction, assume there is some natural number $n$ such that $\neg P(n,\vec{x})$. Then the class given by the formula ``$k$ is a natural number and $\neg P(k,\vec{x})$'' is non-empty and does not have the empty set in it. By Proposition \ref{class_wo}, there is a $\leq_o$-minimal element of this class. Let $n$ be this minimal element, then there is a natural number $m$ such that $n = m\cup\{m\}$
        and hence  $(m,\in) <_o (n, \in)$. By minimality of $n$, we have $P(m,\vec{x})$. Using (\ref{inductive}), the formula $P(n,\vec{x})$ is true, which contradicts our assumption.
\end{proof}

\begin{lemma}
	Let $X$ be an infinite set and $n$ a natural number, then $n <_c X$.
\end{lemma}
\begin{proof}
	Recall that $n <_c X$ iff $n \leq_c X \land \neg(n =_c X)$. Given that $\neg(n =_c X)$ is an immediate consequence of the definition of infinite set, all that remains is to show that for every natural number $n$, the relation $n \leq_c X$ holds.
	We prove it by induction on $n$.
	
	The case $n=0$ is trivial, since $\emptyset$ is an injective function from $0$ to any other set. Assume $n \leq_c X$ and let $f$ be an injective function from $n$ onto $X$. Using that $\neg(n =_c X)$, there must be some $x\in X$ not in the range of $f$. Then the function $f \cup \{(n,x)\}$ is an injective function from $n\cup\{n\}$ onto $X$.
\end{proof}

\begin{corollary}
	Let $A$ be a finite set and $X$ an infinite set, then $A <_c X$.
\end{corollary}
\begin{proof}
	Let $n$ be a natural number such that $A \leq_c n$. Using the previous lemma, we have $n <_c X$ and $A <_c X$ follows immediately.
\end{proof}

Though we are working in an axiomatic system too weak to prove the existence of
the ordinal $\omega$, nevertheless it is handy to have a first order definition
of a similar well order.
\begin{definition}
  A nonempty well order has \emph{type $\omega$} iff  every element
  has an immediate successor, and every element (except the first one) has an
  immediate predecessor.
\end{definition}
It can be seen in $\estf$ that this actually defines an order type (i.e.,
having type $\omega$ is an order isomorphism class) and coincides with being
isomorphic to $\omega$ in case it exists.

The existence of an order of type $\omega$ is apparently stronger than the Weak
Axiom of Infinity. We will actually show under $\estf$ that they are equivalent
by using Hartogs' Lemma. We start with the following lemma.
\begin{lemma}\label{lem:omega-segment}
	If $\langle X,< \rangle$ is an infinite well order that does not have type $\omega$, then there is some $x\in X$ such that the restriction of $<$ to $x{\downarrow} \times x{\downarrow}$ has type $\omega$.
\end{lemma}
\begin{proof}
	Since $\langle X,< \rangle$ does not have type $\omega$, there must be some $x\in X$ that is not the first element of $X$ and satisfies
	\[
	\neg(\exists y\in X \ (\text{Succ}(y,x)) \lor \neg \exists z\in X \ (\text{Succ}(x,z)).
	\]
	We shall see that in both cases $x{\downarrow}$ is infinite. Let us assume 
	\[
	\neg(\exists y\in X \ (\text{Succ}(y,x)),
	\]
	then $x{\downarrow}$ must be infinite. It cannot be empty, because the first element of $X$ belongs to $x{\downarrow}$; and if it was finite with more than one element, it would have a maximum $y$, which would imply $\text{Succ}(y,x)$.
	
	On the other hand, if we assume
	\[
	\neg \exists z\in X \ (\text{Succ}(x,z)),
	\]
	then $X = x{\downarrow} \cup \{x\}$. Given that $X$ is infinite by assumption, it follows that $x{\downarrow}$ must be infinite.
	
	Finally, let us take $Y = \{x\in X \mid x{\downarrow} \text{ is infinite}\}$. This set is non empty by the previous argument, so it has a least element $y$. We will prove that $y{\downarrow}$ has type $\omega$ with the restricted order.
	Recall that we need to show:
	\[
	\forall a\in y{\downarrow} \ ((\forall b\in y{\downarrow} \ \neg (b<a) \ \lor \ \exists b\in y{\downarrow} \ (\text{Succ}(b,a))) \land \exists c\in y{\downarrow} \ (\text{Succ}(a,c)))
	\]
	Let $a\in y{\downarrow}$, since $y$ is the first element whose set of predecessors is infinite, $a{\downarrow}$ is necessarily finite. In particular it is either empty, in which case $a$ is the first element of $y{\downarrow}$, or it has a greatest element $b$ which satisfies $b\in y{\downarrow}$ and Succ$(b,a)$. All that remains to verify is the existence of some $c\in y{\downarrow}$ such that Succ$(a,c)$.
	First let us note that $a<y$ and $\neg\text{Succ}(a,y)$, so the set $\{z\in y{\downarrow} \mid a<z\}$ is non empty. Taking its first element we obtain a $c$ with the desired property.
\end{proof}

\begin{prop}[$\esti$]\label{pr:well-order-type-omega}
	There is a well ordering $\langle A,< \rangle$ of type $\omega$.
\end{prop}
\begin{proof}
	Let $X$ be any infinite set, and take $A = \aleph(X)$. By the previous lemma, it is enough to show that $A$ is not finite. But this is immediate by Lemma~\ref{aleph2}: If it were finite, then we would have $\aleph(X) <_c X$.
\end{proof}

In order to state our main first theorem, we use Peano systems as a substitute for $\omega$.
\begin{definition}[{\cite[Definition 5.1]{Moscho}}]\label{peano}
	A \emph{Peano system} is any structure $(\mathcal{N},z,S)$ which satisfies the following properties:
	\begin{enumerate}[(1)]
		\item $\mathcal{N}$ is a set (the \emph{domain} of the Peano system), and $z\in\mathcal{N}$.
		\item $S$ is a function on $\mathcal{N}$, $S: \mathcal{N}\rightarrow\mathcal{N}$.
		\item $S$ is an injection.
		\item For each $n\in\mathcal{N}$, $S(n) \neq z$.
		\item For each $X\subseteq\mathcal{N}$,
		\[
			(z\in X \land (\forall n\in \mathcal{N} \ (n\in X \implies S(n)\in X))) \implies X = \mathcal{N}.
		\]
	\end{enumerate}
\end{definition}

Note that the existence of a Peano system immediately implies the Weak Axiom of Infinity.
Assuming there is a Peano system, $\estf$ proves that one can define functions recursively (\cite[Theorem 5.6]{Moscho}). From there, it is straightforward to define addition in a given Peano system as $n+z = n$ and $n+S(m) = S(n+m)$ for each $n,m$.


The following lemma highlights the relation between Peano systems, well orderings, and the Weak Axiom of Infinity.

\begin{lemma}
	\label{peano_is_order}
	The following statements are equivalent:
	\begin{enumerate}[(1)]
		\item There is an infinite set (WAI).
		\item There is a well order of type $\omega$.
		\item There is a Peano system.
%		\item Assume $\langle A,< \rangle$ is a well order with type $\omega$ and let $a_0$ be its first element. Then the function that assigns the immediate successor to each element induces a Peano System on $A$.
%		\item Let $(\mathcal{N}, z, S)$ be a Peano system. Define $n<m$ iff $\exists k\in\mathcal{N}{\setminus} \{z\}$ such that $n+k = m$. Then $\langle \mathcal{N},< \rangle$ is a well ordering and it has type $\omega$.
	\end{enumerate}
\end{lemma}

\section{Exponentially decreasing sequences of powersets}

In this section we state and prove in $\esti$ a version of
Theorem~\ref{informal_hartogs_set} for sequences indexed by any Peano
system. In this way, we morally avoid the Replacement instances needed to construct
$\omega$. To avoid cluttering the notation with the successor function $S$ of a
Peano system, we shall refer to $S(n)$ as $n+1$, $S(S(n))$ as $n+2$ and so on.

\begin{theorem}[$\esti$]
	\label{hartogs_set}
	Let $(\mathcal{N},z,S)$ be a Peano system, then there is no sequence $\{X_n\}_{n\in \mathcal{N}}$ such that for all $n\in A$, $\mathcal{P}(X_{n+1}) <_c X_n$.
\end{theorem}
\begin{proof}
	Assume there is such a sequence, 
	$\mathcal{S} := \{X_n\}_{n\in \mathcal{N}}$,
	then $\estf$ proves that its range $\{X_n \mid n\in \mathcal{N}\}$ exists.
	Moreover, for all $X\in\ran(\mathcal{S})$: 
	\[
	\aleph(X) \subset \mathcal{P}(\WO(X)) \subseteq \mathcal{P}(\WO(\union \{X_n \mid n\in \mathcal{N}\})),
	\]
	where both inclusions follow from Definition \ref{aleph}.
	
	\noindent Thus, for all $X\in \{X_n \mid n\in \mathcal{N}\}$, we have $\aleph(X)\in\mathcal{P}(\mathcal{P}(\WO(\union \{X_n \mid n\in \mathcal{N}\})))$. Hence, $\esti$ proves the existence of the sequence $\{\aleph(X_n)\}_{n\in \mathcal{N}}$.
	
	%% It is straightforward to show that for each $n\in \mathcal{N}$,
        %% $\mathcal{P}^{3}(X_{n+3}) <_c X_n$.
        Since applying powersets preserves lax cardinal inequalities, we have
        \[
          \mathcal{P}^3(X_{n+3}) \leq_c \mathcal{P}^2(X_{n+2})\leq_c \mathcal{P}(X_{n+1}) <_c X_{n}.
        \]
        By applying Theorem \ref{thm:partes-cubo}, we conclude
        $\aleph(X_{n+3}) <_c \mathcal{P}^{3}(X_{n+3}) <_c X_n$.
	
	Let us now show that $\aleph(X_{n+3}) <_c \aleph(X_n)$ for all $n\in
        \mathcal{N}$. Since both terms are $\leq_o$-comparable by Proposition~\ref{class_wo}, we have that either
        $\aleph(X_{n+3}) <_c \aleph(X_n)$ or $\aleph(X_n) \leq_c
        \aleph(X_{n+3})$ holds. But the latter statement implies $\aleph(X_n) <_c X_n$, which contradicts Lemma \ref{aleph2}.
	
	Finally, $\{\aleph(X_n) \mid n\in \mathcal{N}\}$
	%$\{\aleph(X_{3n})\}_{n\in\omega}$ 
	is a family of well ordered sets that is not well founded with the $<_c$
        relation and hence not $\leq_o$-well founded, contradicting Proposition~\ref{class_wo}.
\end{proof}

The following theorem is similar to Theorem~\ref{hartogs_set}, but has a
somewhat different flavor, in that it works with the (possibly proper) class of
natural numbers and with a definable sequence of sets. In its statement and
proof,  $n+1$ denotes $n \cup \{n\}$.
\begin{theorem}[$\estf$]
	\label{hartogs_class}
	There is no formula $\Phi(\cdot,\cdot)$  with at least two free
        variables that satisfy both of the following properties:
	\begin{enumerate}[(i)]
		\item \label{nat_define} For each $n\in\omega$ there is exactly one set $X$ that satisfies $\Phi(n,X)$.
		\item \label{dec_seq} For each $n\in\omega$ the following holds: 
		\[
			\Phi(n,X) \land \Phi(n+1,Y) \implies \mathcal{P}(Y) <_c X.
		\] 
	\end{enumerate}
\end{theorem}

\begin{proof}
	By way of contradiction, let $\Phi(\cdot,\cdot)$ be a formula with at
        least two free variables and assume it satisfies (\ref{nat_define}) and (\ref{dec_seq}), we shall take the following class:
	\[
		 \mathcal{C} := \{X \mid \exists Y,\, \exists n\in\omega, \ \Phi(n,Y) \land X = \aleph(Y)\}.

	Let $X_0$ be such a member of $\mathcal{C}$, then there must be some $Y_0$ and $n_0 \in\omega$ such that $X_0 = \aleph(Y_0) \land \Phi(n_0,Y_0)$.
	The assumption (\ref{nat_define}) implies there are $Y_1 , Y_2, Y_3$ such that $\Phi(n_0 + i , Y_i)$ for $i\in\{1,2,3\}$.
	The proof can be finished analogously to that of
        Theorem~\ref{hartogs_set} by applying (\ref{dec_seq}) for $Y_i$.

The previous theorem handles the case of a uniformly
definable sequence of sets. One possible next step in generality is that of a
(meta-)sequence of definable sets, each one given by a separate formula $\Phi_n$
with exactly one free variable, for each natural number $n$ in the
metatheory.

It turns out that a result analogous to Theorem~\ref{hartogs_class} in this
setting fails. To show this, we take advantage of
pointwise definable extensions to obtain an appropriate
meta-sequence of formulas.
\begin{theorem}[{\cite[Theorem~3(3)]{hamkins}}]\label{th:pointwise-extension}
  Every countable model of $\zfc$ has a class forcing extension that is
  pointwise definable.
\end{theorem}

Then our refutation takes this shape:
\begin{theorem}
	Assume $\zfc$ is consistent. There is a scheme of formulas
        $\Phi_n(\cdot)$ $(n\in\mathbb{N})$ with one free variable
        such that for every $n$,
        \begin{itemize}
        \item
          $\estf \vdash \exists! X, \, \Phi_n (X)$; and
        \item
          $\zfc \nvdash \exists X\,\exists Y,\, \Phi_n (X) \land \Phi_{n+1} (Y)
          \land \neg\bigl(\mathcal{P}(Y) <_c X\bigl)$.
        \end{itemize}
\end{theorem}
\begin{proof}
  We start by taking a countable model of $\zfc$. 
  We extend its signature by adding countably many names for constants
  $c_n$, and for each $n\in\N$ we take the sentence $P_n$ to be
  \[
    c_{n+1} \text{ is a natural number} \land 2^{c_{n+1}} < c_n,
  \]
  where exponentiation of natural numbers is construed by its recursive
  definition and hence it is $\Delta^{\ZF}_1$.
  %
  Since $\zfc+\{P_n\}_{n\in\N}$ is finitely satisfiable,
  it has a model $\mathcal{M}_c =
  (M,E,\{ x_n \}_{n\in \N})$ by Compactness (here, $x_n$ is the
  $\mathcal{M}_c$-interpretation of $c_n$). By
  Löwenhein-Skolem, we may assume that $\mathcal{M}_c$ is countable and
  let $\mathcal{M}=(M,E)$ be its reduct to the language of set
  theory. Finally, we extend it to a pointwise definable model $\mathcal{M}'=(M',E')$
  by using Theorem~\ref{th:pointwise-extension}.
  
  Let $\Omega_n$ be a definition of $x_n$ in $\mathcal{M}'$.	
  Then 
  \[
    \Phi_n (X) \defi \bigl((\exists! Y \ \Omega_n(Y))  \land \Omega_n(X)\bigr) \lor
    \bigl(\neg(\exists! Y \ \Omega_n(Y)) \land X=\emptyset\bigr)
  \] 
  satisfies that 
  \[
    \estf \vdash \exists! X \ \Phi_n (X).
  \]
  By construction, we have that for all $x\in M'$,
  \begin{equation}
    \label{eq:Phi-Omega}
    x = x_n  \quad\text{iff}\quad \mathcal{M}' \models \Phi_n[x]  \quad\text{iff}\quad \mathcal{M}'
    \models \Omega_n[x].
  \end{equation}
  Since the formula
  \[
    \varphi(X,Y) \defi \text{$X$ and $Y$ are natural numbers} \land 2^Y < X
  \]
  is $\Delta^{\ZF}_1$ and $\mathcal{M}'$ is a forcing extension of $\mathcal{M}$,
  $\varphi$ is absolute between them. Hence, for each $n\in \N$ the
  following sentence holds in $\mathcal{M}'$:
  \[
    \forall X\, \forall Y,\,  \Omega_n(X) \land \Omega_{n+1}(Y) \implies \varphi(X,Y).
  \]

  Since $\zfc$ proves that $2^Y < X$ is equivalent to $\mathcal{P}(Y) <_c X$
  for $X,Y$ natural numbers, by (\ref{eq:Phi-Omega}) we have
  \[
    \mathcal{M}' \vDash \forall X\, \forall Y,\,  \Phi_n(X) \land \Phi_{n+1}(Y) \implies \mathcal{P}(Y) <_c X
  \]
  for each $n\in\N$, which implies that 
  \[
    \zfc \nvdash \exists X \exists Y, \, \Phi_n(X) \land \Phi_{n+1}(Y) \land
    \neg\bigl(\mathcal{P}(Y) <_c X\bigr).
    \qedhere
  \]
\end{proof}

Although a non-uniform version of Theorem~\ref{hartogs_class} fails, we can
nevertheless isolate the ideas involved in its proof; namely, passing from an
exponentially decreasing sequence of sets to one of well orders. So, essentially
the same proof shows that if the following statement holds (where
$\mathrm{Well}(A)$ is a formula stating ``$A$ is a well-order''),
\begin{statement}
	\label{easier_hartogs_class}
	For every $n\in\mathbb{N}$, let $\Phi_n(\cdot)$ be a formula with one free variable such that for each $n\in\mathbb{N}$, 
	\[
		\estf \vdash \exists!  A, \, \Phi_n ( A) \land 
		\mathrm{Well}(A)
	\]
	then there must be a $k \in \mathbb{N}$ such that 
	\[
	  \estf \vdash \exists A,B, \, \Phi_k (A) \land \Phi_{k+1} (B) \land A \leq_o B,
	\]
\end{statement}
\noindent then we have
\begin{statement}
	\label{easier_inf_classes}
	For every $n\in\mathbb{N}$, let $\Phi_n(\cdot)$ be a formula with one free variable such that for each $n\in\mathbb{N}$, $\estf \vdash \exists! X \ \Phi_n (X)$, then there must be a $k \in \mathbb{N}$ such that 
	\[
	  \estf \vdash \exists X,Y ,\, \Phi_k (X) \land \Phi_{k+1} (Y) \land \neg(\mathcal{P}^3(Y) <_c X).
	\]
\end{statement}
